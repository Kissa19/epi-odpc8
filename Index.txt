    function formatMultilineText(value) {
      const text = escapeHtml(value || '');
      if (!text.trim()) return '<p>ไม่มีรายละเอียดเพิ่มเติม</p>';
      return text.split(/\n{2,}/).map(block => `<p class="mb-4">${block.replace(/\n/g, '<br>')}</p>`).join('');
    }

    function setImagePreview(slot, url) {

    function previewSelectedImage(slot) {
      const input = document.getElementById(`fieldImage${slot}File`);
      const file = input.files && input.files[0];
      if (!file) return setImagePreview(slot, document.getElementById(`fieldImage${slot}Url`).value);
      if (!file.type.startsWith('image/')) {
        input.value = '';
        return showToast('กรุณาเลือกไฟล์รูปภาพเท่านั้น', true);
      }
      if (file.size > 5 * 1024 * 1024) {
        input.value = '';
        return showToast('ขนาดรูปต้องไม่เกิน 5 MB ต่อรูป', true);
      }
      const reader = new FileReader();
      reader.onload = e => setImagePreview(slot, e.target.result);
      reader.readAsDataURL(file);
    }

    function readImageFile(slot) {
      const input = document.getElementById(`fieldImage${slot}File`);
      const file = input.files && input.files[0];
      if (!file) return Promise.resolve({ dataUrl: '', name: '' });
      if (!file.type.startsWith('image/')) return Promise.reject(new Error('กรุณาเลือกไฟล์รูปภาพเท่านั้น'));
      if (file.size > 5 * 1024 * 1024) return Promise.reject(new Error('ขนาดรูปต้องไม่เกิน 5 MB ต่อรูป'));
      return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = e => resolve({ dataUrl: e.target.result, name: file.name });
        reader.onerror = () => reject(new Error('อ่านไฟล์รูปภาพไม่สำเร็จ'));
        reader.readAsDataURL(file);
      });
    }

    async function saveContent(e) {
      e.preventDefault();
      if (isSavingContent) return;
      const btn = document.getElementById('saveBtn');
      const original = btn.innerHTML;
      isSavingContent = true;
      setSavingOverlay(true);
      setEditorFormDisabled(true);
      btn.disabled = true;
      btn.innerHTML = '<i class="fa-solid fa-spinner fa-spin mr-2"></i>กำลังบันทึก...';
      let image1 = { dataUrl: '', name: '' };
      let image2 = { dataUrl: '', name: '' };
      try {
        if (document.getElementById('fieldCategory').value === 'news') {
          image1 = await readImageFile(1);
          image2 = await readImageFile(2);
        }
      } catch (err) {
        btn.disabled = false;
        btn.innerHTML = original;
        isSavingContent = false;
        setSavingOverlay(false);
        setEditorFormDisabled(false);
        return showToast(err.message, true);
      }
      const payload = {
        id: document.getElementById('fieldId').value,
        category: document.getElementById('fieldCategory').value,
        status: document.getElementById('fieldStatus').value,
        title: document.getElementById('fieldTitle').value,
        subtitle: document.getElementById('fieldSubtitle').value,
        description: document.getElementById('fieldDescription').value,
        url: document.getElementById('fieldUrl').value,
        icon: getSelectedIconValue(),
        accent: document.getElementById('fieldAccent').value,
        dateText: document.getElementById('fieldDateText').value,
        sortOrder: document.getElementById('fieldSortOrder').value,
        featured: document.getElementById('fieldFeatured').checked,
        image1Url: document.getElementById('fieldImage1Url').value,
        image2Url: document.getElementById('fieldImage2Url').value,
        image1DataUrl: image1.dataUrl,
        image1Name: image1.name,
        image2DataUrl: image2.dataUrl,
        image2Name: image2.name
      };
      try {
        await callServer('saveContent', [payload, adminPassword]);
        clearEditorDraft();
        editorHasUnsavedChanges = false;
        closeModal('editorModal');
        await reloadAdminData();
        showToast('บันทึกข้อมูลเรียบร้อยแล้ว');
      } catch (err) {
        showToast(err.message, true);
      } finally {
        isSavingContent = false;
        setSavingOverlay(false);
        setEditorFormDisabled(false);
        btn.disabled = false;
        btn.innerHTML = original;
      }
    }

    async function deleteById(id) {
      if (!adminPassword) return openModal('loginModal');
      const item = items.find(x => x.id === id);
      const ok = confirm(`ยืนยันการลบรายการ: ${item ? item.title : id} ?`);
      if (!ok) return;
      try {
        await callServer('deleteContent', [id, adminPassword]);
        closeModal('editorModal');
        await reloadAdminData();
        showToast('ลบข้อมูลเรียบร้อยแล้ว');
      } catch (err) {
        showToast(err.message, true);
      }
    }

    function renderSearchResults() {
      const q = (document.getElementById('globalSearch').value || '').toLowerCase().trim();
      const source = q ? items.filter(item => [item.title, item.subtitle, item.description].join(' ').toLowerCase().includes(q)) : items.slice(0, 12);
      document.getElementById('searchResults').innerHTML = source.length ? source.map(item => {
        const sec = sections.find(s => s.key === item.category) || { title: item.category };
        const openAction = item.category === 'news' ? `<button onclick="closeModal('searchModal'); openNewsDetail('${escapeAttr(item.id)}')" class="inline-flex items-center gap-2 text-sm font-semibold text-sky-700 hover:text-sky-900">อ่านรายละเอียด<i class="fa-solid fa-book-open text-xs"></i></button>` : renderOpenLink(item, 'เปิด');
        return `<div class="py-4 flex gap-4"><div class="w-11 h-11 rounded-2xl bg-sky-50 text-sky-600 flex items-center justify-center"><i class="fa-solid ${escapeAttr(item.icon || 'fa-file-lines')}"></i></div><div class="flex-1"><p class="font-bold text-slate-900">${escapeHtml(item.title)}</p><p class="text-sm text-slate-400">${escapeHtml(sec.title)}</p><p class="text-sm text-slate-500 mt-1">${escapeHtml(item.description || '')}</p><div class="mt-2">${openAction}</div></div></div>`;
      }).join('') : '<div class="py-8 text-center text-slate-400">ไม่พบข้อมูลที่ค้นหา</div>';
    }

    function openModal(id) {
      const modal = document.getElementById(id);
      if (!modal) return;
      modal.classList.remove('hidden');
      modal.classList.add('flex');
      if (id === 'editorModal') setTimeout(restoreEditorDraftIfNeeded, 0);
    }

    function closeModal(id, force = false) {
      if (id === 'editorModal' && !force) {
        if (isSavingContent) return showToast('กำลังบันทึกข้อมูล กรุณารอสักครู่', true);
        if (editorHasUnsavedChanges && !confirm('มีข้อมูลที่ยังไม่ได้บันทึก ต้องการปิดหน้าต่างนี้หรือไม่? ระบบจะเก็บร่างไว้ชั่วคราวในเบราว์เซอร์')) return;
      }
      const modal = document.getElementById(id);
      if (!modal) return;
      modal.classList.add('hidden');
      modal.classList.remove('flex');
    }

    function bindModalBackdrop() {
      document.querySelectorAll('.fixed.inset-0').forEach(modal => modal.addEventListener('click', e => {
        if (e.target === modal) {
          if (modal.id === 'editorModal') return; // ป้องกันคลิกพื้นหลังแล้วฟอร์มหายระหว่างกรอก/อัปโหลด
          closeModal(modal.id);
        }
      }));
    }

      document.querySelectorAll('#contentForm input, #contentForm textarea, #contentForm select, #contentForm button').forEach(el => {

    function bindEditorDraftListeners() {
      const form = document.getElementById('contentForm');
      if (!form) return;
      form.addEventListener('input', rememberEditorDraft);
      form.addEventListener('change', rememberEditorDraft);
      window.addEventListener('beforeunload', e => {
        if (isSavingContent || editorHasUnsavedChanges) {
          e.preventDefault();
          e.returnValue = '';
        }
      });
    }

    function rememberEditorDraft() {
      const modal = document.getElementById('editorModal');
      if (!modal || modal.classList.contains('hidden')) return;
      editorHasUnsavedChanges = true;
      const draft = {
        id: document.getElementById('fieldId')?.value || '',
        category: document.getElementById('fieldCategory')?.value || '',
        status: document.getElementById('fieldStatus')?.value || '',
        title: document.getElementById('fieldTitle')?.value || '',
        subtitle: document.getElementById('fieldSubtitle')?.value || '',
        description: document.getElementById('fieldDescription')?.value || '',
        url: document.getElementById('fieldUrl')?.value || '',
        icon: getSelectedIconValue(),
        customIcon: document.getElementById('fieldIconCustom')?.value || '',
        accent: document.getElementById('fieldAccent')?.value || '',
        dateText: document.getElementById('fieldDateText')?.value || '',
        sortOrder: document.getElementById('fieldSortOrder')?.value || '',
        featured: document.getElementById('fieldFeatured')?.checked || false,
        image1Url: document.getElementById('fieldImage1Url')?.value || '',
        image2Url: document.getElementById('fieldImage2Url')?.value || ''
      };
      sessionStorage.setItem('epiEditorDraft', JSON.stringify(draft));
    }

    function clearEditorDraft() {
      sessionStorage.removeItem('epiEditorDraft');
    }

    function restoreEditorDraftIfNeeded() {
      const raw = sessionStorage.getItem('epiEditorDraft');
      if (!raw) return;
      let draft;
      try { draft = JSON.parse(raw); } catch (e) { return; }
      const currentId = document.getElementById('fieldId')?.value || '';
      if (currentId && draft.id && currentId !== draft.id) return;
      if (!draft.title && !draft.description && !draft.url) return;
      if (!confirm('พบข้อมูลร่างที่ยังไม่ได้บันทึก ต้องการกู้คืนข้อมูลที่กรอกไว้หรือไม่?')) return;
      ['fieldId','fieldCategory','fieldStatus','fieldTitle','fieldSubtitle','fieldDescription','fieldUrl','fieldAccent','fieldDateText','fieldSortOrder','fieldImage1Url','fieldImage2Url'].forEach(id => {
        const key = id.replace('field','');
        const camel = key.charAt(0).toLowerCase() + key.slice(1);
        if (document.getElementById(id) && draft[camel] !== undefined) document.getElementById(id).value = draft[camel];
      });
      document.getElementById('fieldFeatured').checked = !!draft.featured;
      setIconValue(draft.icon || draft.customIcon || 'fa-file-lines');
      setImagePreview(1, draft.image1Url || '');
      setImagePreview(2, draft.image2Url || '');
      toggleNewsImageFields();
    }

    function showToast(message, isError = false) {
      const toast = document.getElementById('toast');
      toast.textContent = message;
      toast.className = `fixed bottom-5 right-5 z-[120] rounded-2xl px-5 py-4 shadow-soft text-sm ${isError ? 'bg-rose-600 text-white' : 'bg-slate-900 text-white'}`;
      toast.classList.remove('hidden');
      setTimeout(() => toast.classList.add('hidden'), 3500);
    }

    function escapeHtml(value) { return String(value || '').replace(/[&<>'"]/g, c => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', "'": '&#39;', '"': '&quot;' }[c])); }
    function escapeAttr(value) { return escapeHtml(value).replace(/`/g, '&#96;'); }
